## 🍡第一部分：Node基础操作

### 1.文件系统

```js
// ============++ 同步 ++============

// 1.同步读取文件
fs.readFileSync('input.txt').toString()

// 2.同步写入文件
fs.writeFileSync(file, data)

// 3.同步追加内容
fs.appendFileSync(file, data)

// 4.同步读取目录
fs.readdirSync(path)

// 5.同步创建目录
fs.mkdirSync(path, [option]) //  options 参数一 是否以递归的方式创建目录，默认false |参数二： mode 设置目录权限，默认0777

// 6.同步删除目录
fs.rmdirSync(path)

// 7.获取文件信息
fs.statSync('路径')

// 8.删除文件
fs.unlinkSync(path)



// ============++ 异步 ++============

// 1.异步读取文件
fs.readFile('input.txt', (err, data) => {
   if (err) console.error(err);
   console.log("异步读取: " + data.toString());
})

// 其余的都是一个道理，去除掉Sync，加个回调函数
```

### 2.启动http服务

```javascript
//导入HTTP模块
let http = require('http')
 
//定义一个要监听的端口
const PORT = 8989;
 
// 一个处理请求和发送响应的函数
const handleRequest = (request, response) => {
    response.end('Hello World!!!')
}

// 创建服务器
let server = http.createServer(handleRequest);
 
// 启动服务
server.listen(PORT, () => {
    console.log("Server listening on: http://localhost:%s", PORT);
})
```

### 3.快速启动一个代理静态资源的HTTP服务

::: tip
中文目录会有代理不上的地方（一般也不用纯Node代理，直接用第三方框架代理）
:::

```javascript
let http = require('http')
let fs = require('fs')
let url = require('url')
let path = require('path')

let server = http.createServer(function(request, response){
    //获取输入的url解析后的对象
    let pathObj = url.parse(request.url, true);
    //static文件夹的绝对路径
    let staticPath = path.resolve(__dirname, 'dist')
    //获取资源文件绝对路径
    let filePath = path.join(staticPath, pathObj.pathname)
    // 如果是根目录的情况下的访问，需要拼接上 /index.html
    if (filePath.charAt(filePath.length-1)=="/"){
      filePath+="index.html";
    }
    //异步读取file
    fs.readFile(filePath, 'binary', function(err, fileContent){
        if (err) {
            console.log('404')
            response.writeHead(404, 'not found')
            response.end('<h1>404 Not Found</h1>')
        } else {
            console.log('ok')
            response.write(fileContent, 'binary')
            response.end()
        }
    })
})

server.listen(9999)
console.log('visit http://localhost:9999')
```

## 🍭第二部分：Node常用的组合操作

### 1.复制某文件夹下所有的文件到另一个文件夹

```js
const fs = require('fs')

// 复制文件夹
function exists (src, dst) {
  //测试某个路径下文件是否存在
  fs.exists(dst, function(exists){
      if (exists) {//不存在
          copy(src,dst);
      } else {//存在
          fs.mkdir(dst,function(){//创建目录
              copy(src,dst)
          })
      }
  })
}

// 复制文件夹的辅助函数
function copy (src, dst) {
  //读取目录
  let paths = fs.readdirSync(src);
  paths.forEach(function(path){
      var _src=src+'/'+path;
      var _dst=dst+'/'+path;
      var readable;
      var writable;
      fs.stat(_src,function(err,st){
          if(err){
              throw err;
          }
          
          if(st.isFile()){
              readable=fs.createReadStream(_src);//创建读取流
              writable=fs.createWriteStream(_dst);//创建写入流
              readable.pipe(writable);
          }else if(st.isDirectory()){
              exists(_src,_dst,copy);
          }
      });
  });
}

// 使用演示 将 a 文件夹下 所有的文件，复制到 b文件夹下
exists('./a', './b')
```
